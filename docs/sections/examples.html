Examples

Below are several example scripts demonstrating the features and functionalities of the Pithon programming language. These examples cover basic operations, function definitions, set manipulations, and more.

Basic Arithmetic Operations


# basic_arithmetic.pi

let a = 10;
let b = 5;

let sum = a + b;
let difference = a - b;
let product = a * b;
let quotient = a / b;

display("Sum:", sum);           # Output: Sum: 15
display("Difference:", difference); # Output: Difference: 5
display("Product:", product);       # Output: Product: 50
display("Quotient:", quotient);     # Output: Quotient: 2.0

Using Built-In Math Constants


# math_constants.pi

let circumference = 2 * Math.pi * radius;
let area = Math.pi * (radius ^ 2);

display("Circumference:", circumference);
display("Area:", area);

Defining and Using Functions


# define_function.pi

# Define a function to calculate the square of a number
let square(x) = x ^ 2;

# Use the function
let result = square(4);

display("Square of 4:", result);  # Output: Square of 4: 16

Applying a Function to a Set


# apply_function.pi

# Define a set
let numbers = {1, 2, 3, 4, 5};

# Define a function that squares its input
let square(x) = x ^ 2;

# Apply the function to each element of the set
let squared_numbers = apply(square, numbers);

display("Original Numbers:", numbers);         # Output: Original Numbers: {1, 2, 3, 4, 5}
display("Squared Numbers:", squared_numbers);  # Output: Squared Numbers: {1, 4, 9, 16, 25}

Using Conditional Statements


# conditional_statement.pi

let number = 10;

if (number > 0){
    display(number, "is positive.");
}
else {
    display(number, "is not positive.");
}

Looping with For Loops


# for_loop.pi

# Print numbers from 1 to 5
for i in 1 <+> 6 {
    display("Number:", i);
}
